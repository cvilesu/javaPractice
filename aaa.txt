public class Y1 {

	public static void main(String[] args) {
	Scanner Input = new Scanner(System.in);	
	System.out.println("Ukucaj broj fibonacci brojeva u nizu: ");
	int bfbn=Input.nextInt();
	int[] fibniz = new int[bfbn];
	int brojac=0;
	fibniz[0]=0;
	fibniz[1]=1;
	for(brojac=2;brojac<bfbn;brojac++)
		{
			fibniz[brojac]=fibniz[brojac-2]+fibniz[brojac-1];
		}
	System.out.print("Fibonacci niz cine: ");
	for(brojac=0;brojac<bfbn;brojac++)
		{
		System.out.print(fibniz[brojac]+", ");
		}
	}

}
